BENCHMARK(single_digit_ints_FAST_FLOAT, n) {
  double value{};
  constexpr std::size_t kSelectionMask = 8 - 1;
  for (std::size_t i = 0; i < n; i++) {
    std::string& input = singleDigitIntValues[i & kSelectionMask];
    char* b = input.data();
    char* e = b + input.size();
    fast_float::from_chars(b, e, value);
  }
}
