BENCHMARK(random_input_FAST_FLOAT, n) {
  double value{};
  for (unsigned int i = 0; i < n; ++i) {
    std::string& input = randomValues[fastFloatRandomInputIndex];
    fastFloatRandomInputIndex += 1;
    char* b = input.data();
    char* e = b + input.size();
    fast_float::from_chars(b, e, value);
  }
}
