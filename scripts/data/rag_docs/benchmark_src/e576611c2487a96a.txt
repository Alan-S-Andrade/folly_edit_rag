BENCHMARK(ipv6_try_from_string_valid, iters) {
  while (iters--) {
    auto maybeIp =
        IPAddressV6::tryFromString("2803:6082:18e0:2c49:1a23:9ee0:5c87:9800");
    CHECK(maybeIp.hasValue());
    doNotOptimizeAway(maybeIp);
    doNotOptimizeAway(maybeIp.value());
  }
}
